### hermes-worker.service
# /etc/systemd/system/hermes-worker.service
[Unit]
Description=Crypto Tuber Ranked Hermes worker
Requires=docker.service
After=docker.service network-online.target
Wants=network-online.target

[Service]
Type=oneshot
WorkingDirectory=/opt/crypto-tuber-ranked
RemainAfterExit=yes
ExecStart=/usr/bin/docker compose up -d hermes-worker
ExecStop=/usr/bin/docker compose down
TimeoutStartSec=0

[Install]
WantedBy=multi-user.target
### callscore-daily-pipeline.service
# /etc/systemd/system/callscore-daily-pipeline.service
[Unit]
Description=CallScore daily slow transcript/data pipeline
After=network-online.target docker.service postgresql.service
Wants=network-online.target

[Service]
Type=oneshot
WorkingDirectory=/opt/crypto-tuber-ranked
Environment=DATABASE_PROVIDER=postgres
Environment=NODE_ENV=production
Environment=HH_READ_API_BASE=https://ops-bridge.call-score.com/api/read
EnvironmentFile=-/opt/crypto-tuber-ranked/.env.hermes
ExecStart=/usr/bin/npm run pipeline:daily -- --write --read-api-base https://ops-bridge.call-score.com/api/read --transcript-limit 25 --transcript-concurrency 1 --transcript-gap-ms 20000 --limit-creators 250 --limit-videos 10 --since-days 45 --extract-limit 50 --match-limit 500 --match-batch-size 100
Nice=10
IOSchedulingClass=best-effort
IOSchedulingPriority=7
